# 자료구조란? 
**자료구조의 종류?** 스택, 큐, 리스트, 사전, 그래프, 트리
**프로그램**= 자료구조 + 알고리즘

```
#define MAX_ELEMENTS 100
int scores[MAX_ELEMENTS]; // 배열 scores가 자료구조!! 
int get_max_score(int n) // 변수 largest를 첫 번째 요소로 초기화, 나머지 요소를 순차적으로 비교
{
    int i, largest;
    largest = scores[0];
    for (i=1; i<n; i++){
            if (scores[i] > largest){
                       largest=scores[i];
            }
    }
    return largest;
}
